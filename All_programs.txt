16) Write a program to demonstrate the difference between pre-increment and post-increment.

#include <stdio.h>

int main() {
    int x, y;

    // Pre-increment demonstration
    x = 10;
    y = ++x;  // Increment x, then assign it to y
    printf("Pre-increment:\n");
    printf("x = %d, y = %d  (x was incremented before assignment)\n\n", x, y);

    // Post-increment demonstration
    x = 10;
    y = x++;  // Assign x to y, then increment x
    printf("Post-increment:\n");
    printf("x = %d, y = %d  (x was incremented after assignment)\n", x, y);

    return 0;
}

    printf("Sum = %d\n", sum);
    printf("Difference = %d\n", difference);

}

        printf("Invalid input! Please enter valid integers.\n");

}

    char name[50];
    int age;
    float percentage;
};

int main() {
    struct Student s;

    // Prompt for and read input
    printf("Enter student's name: ");
    scanf("%49s", s.name);  // Limit to 49 characters to avoid buffer overflow

    printf("Enter student's age: ");
    scanf("%d", &s.age);

    printf("Enter student's percentage: ");
    scanf("%f", &s.percentage);

    // Display the data with formatted output
    printf("\n--- Student Details ---\n");
    printf("Name       : %s\n", s.name);
    printf("Age        : %d years\n", s.age);
    printf("Percentage : %.2f%%\n", s.percentage);

    return 0;
}
 21) Write a program to input radius of a circle and print area and circumference.

#include <stdio.h>

#define PI 3.14159

int main() {
    double radius, area, circumference;

    printf("Enter the radius of the circle: ");
    if (scanf("%lf", &radius) != 1) {
        printf("Invalid input!\n");
        return 1;
    }

    area = PI * radius * radius;
    circumference = 2 * PI * radius;

    printf("\n--- Circle Metrics ---\n");
    printf("Radius         : %.2f units\n", radius);
    printf("Area           : %.2f square units\n", area);
    printf("Circumference  : %.2f units\n", circumference);

    return 0;
}

22) Write a program to input temperature in Celsius and convert it into Fahrenheit.

#include <stdio.h>

int main() {
    float celsius, fahrenheit;

    // Prompt user for input
    printf("Enter temperature in Celsius: ");
    scanf("%f", &celsius);

    // Convert Celsius to Fahrenheit
    fahrenheit = (celsius * 9.0 / 5.0) + 32;

    // Display the result
    printf("%.2f Celsius is equal to %.2f Fahrenheit\n", celsius, fahrenheit);

    return 0;
}
